    ```nix
                ./modules/noctalia.nix
              ];
            };
          };
        };
    }
    ```

    Create src/generator/flake.rs with FlakeContext struct:
    ```rust
    #[derive(Template)]
    #[template(path = "flake.nix")]
    struct FlakeContext {
        hostname: String,
        username: String,
        nixpkgs_ref: String,
    }

    pub struct FlakeGenerator;

    impl Generator for FlakeGenerator {
        fn generate(&self, config: &UserConfig) -> Result<GeneratedFile> {
            let context = FlakeContext {
                hostname: config.hostname.clone(),
                username: config.username.clone(),
                nixpkgs_ref: "nixos-unstable".to_string(),
            };
            let content = context.render()?;
            Ok(GeneratedFile {
                path: PathBuf::from("flake.nix"),
                content,
            })
        }

        fn output_path(&self, _config: &UserConfig) -> PathBuf {
            PathBuf::from("flake.nix")
        }
    }
    ```
  </action>
  <verify>FlakeContext renders flake.nix template without Askama errors</verify>
  <done>flake.nix template exists and FlakeGenerator implements Generator trait</done>
