    ```rust
    /// Generate all configuration files from user configuration
    ///
    /// # Arguments
    /// * `config` - UserConfig derived from WizardState
    /// * `output_dir` - Directory to write generated files
    ///
    /// # Returns
    /// Vector of paths to generated files
    ///
    /// # Errors
    /// Returns GeneratorError if any template fails to render,
    /// validation fails, or file write fails
    pub fn generate_all(
        config: &UserConfig,
        output_dir: &Path,
    ) -> Result<Vec<PathBuf>, GeneratorError> {
        let mut generated_paths = Vec::new();

        for generator in all_generators() {
            // Generate the file content
            let generated = generator.generate(config)?;

            // Validate before writing
            generator.validate(&generated.content)?;

            // Write atomically
            let output_path = output_dir.join(&generated.path);
            write_config_atomically(&output_path, &generated.content)?;

            generated_paths.push(generated.path.clone());

            // Also write modules to subdirectory
            if generated.path.starts_with("modules/") {
                let modules_dir = output_dir.join("modules");
                write_config_atomically(&output_dir.join(&generated.path), &generated.content)?;
            }
        }

        Ok(generated_paths)
    }
    ```

    Handle the modules/ path correctly - write modules to output_dir/modules/.
